\section{Was ist \LaTeX?}

\begin{frame}
\frametitle{Was ist \LaTeX? (1)}
\enquote{Leslie Lamport entwickelte LaTeX als strukturierte, high-level Schnittstelle zu TeX. Technisch gesehen ist LaTeX ein großes Makro-Paket, das zusätzlich zu TeX geladen wird. Ein Vorteil der Verwendung von LaTeX anstelle von "einfachem" TeX ist, dass LaTeX viele stilistische Details automatisch übernimmt; außerdem haben viele Leute Zusatzpakete für LaTeX geschrieben, die häufige Satzprobleme lösen. Insbesondere hat die American Mathematical Society die Entwicklung eines Pakets namens \enquote{amsmath} gefördert, das den Schriftsatz komplizierter mathematischer Ausdrücke vereinfacht.} \\ (https://wiki.lyx.org/FAQ/LaTeX)
\end{frame}

\begin{frame}
\frametitle{Was ist \LaTeX? (2)}
\enquote{Donald E. Knuth, ein Mathematiker und Informatiker, entwickelte das TeX-Satzsystem "für die Erstellung schöner Bücher - und insbesondere für Bücher, die viel Mathematik enthalten". Seine brillante Arbeit war ein voller Erfolg, und einige Varianten von TeX werden heute von den meisten professionellen Mathematikern verwendet. Wenn Sie ein zufälliges Mathematikbuch nehmen, das in den letzten fünf Jahren veröffentlicht wurde, stehen die Chancen gut, dass es mit TeX formatiert wurde.} \\ (https://wiki.lyx.org/FAQ/TeX)
\end{frame}

\section{Geschichte von \TeX \ und \LaTeX}
\begin{frame}
  \frametitle{Historie von \TeX}
\begin{minipage}[c]{0.48\linewidth}
          \begin{figure}
           \centering
           \includegraphics[width=3.5cm]{img/knuth}
           \caption*{{\tiny Donald E. Knuth \\ (http://bit.ly/2XiPd9M)}}
           \end{figure}
\end{minipage}
\begin{minipage}[c]{0.48\linewidth}
              \begin{itemize}
  \item \TeX (gr: "Kunst", sprich  \begin{IPA} [tEC] \end{IPA})
  \item Entwickelt ab 1977 an der Stanfort University von Donald E. Knuth
  \item Ziel: Erstellen von formschönen Zeichensätzen und mathematischen Formeln
  \item Nachteil: sehr kompliziert
  \end{itemize}
\end{minipage}
\end{frame}
\begin{frame}
\frametitle{Historie von \LaTeX}
\begin{minipage}[c]{0.48\linewidth}
          \begin{figure}
           \centering
           \includegraphics[width=3.5cm]{img/lamport}
           \caption*{{\tiny Leslie Lamport \\ (http://lamport.org)}}
           \end{figure}
\end{minipage}
\begin{minipage}[c]{0.48\linewidth}
              \begin{itemize}
  \item \LaTeX (sprich  \begin{IPA} [la:tEC] \end{IPA})
  \item Weiterentwicklung ab 1982 durch Leslie Lamport
  \item Interpreter zwischen Anwender und \TeX
  \item Ziel: Weniger aufwändiger Textsatz als mit \TeX
  \end{itemize}
\end{minipage}
\end{frame}
\section{WYSIWYG vs WYSIWYAF}
\begin{frame}
\frametitle{Word vs \LaTeX}
\begin{tabular}{c|c}
Word (WYSIWYG) & \LaTeX \ (WYSIWYAF)\\ 
\hline 
+ Einfache Nutzung & - Schwieriger Einstieg \\ 
+ Tabellen, Grafiken, ... & - Unübersichtlich \\ 
+ Kein verwirrender Quellcode & + verwirrender Quellcode \\ 
- Versionsstabilität & + Versionsstabilität \\ 
- Eine Datei & + Mehrere Dateien \\
\end{tabular} 
\end{frame}
